\begin{problem}{Complete the sequence}

  \underline{\hspace{2em}}$

  \item $10000, 5000, 2500, \underline{\hspace{3em}}, \underline{\hspace{3em}}$

  \item $1, 1, 2, 3, 5, 8, 13, \underline{\hspace{2em}},

 \end{enumerate}
